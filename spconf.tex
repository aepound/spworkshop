% Record version of MATLAB/Octave
a = ver('octave');

    a = ver('matlab');
end

fprintf(fid, [a.Name ' version ' a.Version]);
fclose(fid);

% Record computer architecture.

fclose(fid);

% Record date of run.

fprintf(fid, datestr(now, 'dd/mm/yyyy'));
fclose(fid);
